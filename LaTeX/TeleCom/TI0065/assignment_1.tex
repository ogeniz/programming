\section{Questão 2.4}

Determine a largura de feixe de meia potência (LFMP) e a largura de
feixe entre nulos (LFEN), em radianos e em graus, para a seguinte
intensidade de radiação normalizada: $U(\theta) = \cos^{2}{\theta}$,
para

$ \left\{\begin{array}{l}
           0 \leq \theta \leq 90\degree \\
           0 \leq \phi \leq 360\degree
         \end{array}\right.
       $


       \begin{enumerate}[i]
       \item LFMP. Seja $U(\theta_{h}) = \dfrac{1}{2}$ $\Rightarrow$
         $\cos^{2}{\theta_{h}} = \dfrac{1}{2}$ $\Rightarrow$
         $\theta_h = \cos^{-1}\left(\sqrt{\dfrac{1}{2}}\right)$
         $\Rightarrow$ $\theta_{h} = \dfrac{\pi}{4}$. A função
         $U(\theta)$ é simétrica em relação ao máximo em $\theta = 0$,
         assim LFMP $=$
         $2\theta_{h} = \dfrac{\pi}{2} \, \mathrm{ou} \,
         90\si{\degree}$
       \item LFEN. Seja $U(\theta_{n}) = 0$ $\Rightarrow$ $\theta_{n}$
         $=$ $\cos^{-1}\left(0\right)$ $\Rightarrow$ $\theta_{n}$ $=$
         $\dfrac{\pi}{2}$. Pela mesma razão do item anterior LFEN $=$
         $2\theta_{n}$ $=$ $\pi \, \mathrm{ou} \, 180\si{\degree}$
       \end{enumerate}


       \section{Questão 2.5}

       Determine a largura de feixe de meia potência (LFMP) e a
       largura de feixe entre nulos (LFEN), em radianos e em graus,
       para a seguinte intensidade de radiação normalizada:
       $U(\theta) = \cos{2\theta}\cos{3\theta}$, para

       $ \left\{\begin{array}{l}
                  0 \leq \theta \leq 90\degree \\
                  0 \leq \phi \leq 360\degree
                \end{array}\right.
              $


              \begin{enumerate}[i]
              \item LFMP. Seja $U(\theta_{h}) = \dfrac{1}{2}$
                $\Rightarrow$
                \fbox{$\cos{2\theta_{h}}\cos{3\theta_{h}} -
                  \dfrac{1}{2} = 0$}. A equação, em destaque, é
                transcendental. Sendo necessário o uso de técnicas
                numéricas para encontrar suas raízes. Usando a função
                solve de um {\bfseries{CAS}}(\emph{Computer Algebra
                  System}), $\theta_{h} \approx 0,3049\si{\radian}$ ou
                $\approx$ $17,4711\si{\degree}$. A função $U(\theta)$
                é simétrica em relação ao máximo em $\theta = 0$,
                assim
                $ \mathrm{LFMP} \, = 2\theta_{h} \approx
                0,6098\si{\radian} \Rightarrow \theta_{n} \approx
                34,94\si{\degree}$.
    
              \item LFEN. Seja $U(\theta_{n}) = 0$ $\Rightarrow$
                \fbox{$\cos{2\theta_{n}}\cos{3\theta_{n}} = 0$}. Isto
                nos leva a
                duas soluções para $\theta_{n}$.\\
                $ \left\{\begin{array}{l} \cos{2\theta_{n}} = 0
                    \Rightarrow \theta_{n} =
                           \dfrac{1}{2}\cos^{-1}\left(0\right) = \dfrac{\pi}{4}\\
                           \\ \cos{3\theta_{n}} = 0 \Rightarrow
                           \theta_{n} =
                           \dfrac{1}{3}\cos^{-1}\left(0\right) =
                           \dfrac{\pi}{6}
                         \end{array}\right.
                       $


                       O menor valor de $\theta_{n}$ é o LFEN, assim,
                       seguindo a o mesmo motivo do item anterior LFEN
                       $=$ $2\theta_{n}$ $=$ $\dfrac{\pi}{6}$
                       $\Rightarrow$ $\theta_{n}$ $=$ $\dfrac{\pi}{3}$
                       ou $60\si{\degree}$
                     \end{enumerate}

\section{Questão 2.7}
A potência radiada por uma antena sem perda é $14\si{\watt}$. As
características direcionais da antena são representadas pelas
seguintes intensidades de radiação:

$ \left\{\begin{array}{l|l}
           U(\theta)= B_0\cos^2\theta & \text{(watts/unidades de ângulo sólido)} \\
           U(\theta)= B_0\cos^3\theta & (0 \leq \theta \leq \frac{\pi}{2} , 0 \leq \phi \leq 2\pi) \\
         \end{array}\right.
       $

       \vspace*{0.5cm}Para cada uma, determine:

       \begin{enumerate}[(a)]

       \item A máxima densidade de potência (em
         $\si{\watt/\meter^{2}}$) a uma distância de
         1.000$\si{\meter}$ (admita que esta é uma distância de campo
         distante). Especifique o ângulo em que isso ocorre.
       \item Valores exato e aproximado para o ângulo sólido de feixe
         $\Omega_A$.
       \item Diretividade, exata e aproximada, da antena (adimensional
         e em dB).
       \item Ganho, exato e aproximado, da antena (adimensional e em
         dB)
       \end{enumerate}


       \begin{enumerate}[a.]
       \item Máxima densidade de potência $[\si{\watt/\meter^{2}}]$
         \begin{enumerate}[i.]
         \item $U(\theta)$ $=$ $B_{0}\cos^{2}{\theta}$. \\ A potência
           radiada é dada por $P_{rad}$ $=$
           $\displaystyle \oiint U d\Omega$, com
           $P_{rad} = 14\si{\watt}$ e
           $d\Omega = \sin{\theta} d\theta d\phi$. Assim, temos:

           \begin{eqnarray*}
             14 & = &\int^{2\pi}_{0}\int^{\frac{\pi}{2}}_{0}
                      B_{0}\cos^{2}{\theta}\sin{\theta} d\theta d\phi \\
             14 & = & 2B_{0}\pi \int^{\frac{\pi}{2}}_{0} \cos^{2}{\theta}\sin{\theta} d\theta
           \end{eqnarray*}

           
           Fazendo a seguinte substituição de variável

           $ \left\{\begin{array}{l}
                      u = \cos{\theta} \\
                      -du = \sin{\theta} d\theta
                    \end{array}\right.
                  $

                  e dos limites, temos

                \begin{eqnarray*}
                  14 & = & (-1)2B_{0}\pi \int^{0}_{1}u^{2} du \\
                     & = & 2B_{0}\pi \int^{1}_{0}u^{2} du \\                  
                     & = & 2B_{0}\pi \left(\dfrac{u^{3}}{3}\right)^{1}_{0} \\
                  14 & = & \dfrac{2B_{0}\pi}{3} \\
                  B_{0} & = & \dfrac{21}{\pi}
                \end{eqnarray*}

                A densidade de potência é dado por
                $W_{rad} = \dfrac{U(\theta,\phi)}{r^{2}}$, com
                $U(\theta,\phi) = \dfrac{21}{\pi}\cos^{2}{\theta}$,
                para seu valor máximo temos $\theta = k\pi$, com
                $k = 2n$ e $n \in \mathbb{Z}$, assim
                $\cos^{2}{\theta}$ $=$ $1$.

                \begin{eqnarray*}
                  W_{M} & = & \dfrac{21}{\pi}\left(
                              \dfrac{1}{10^{3}}\right)^{2} \\
                  W_{M} & = & \dfrac{21}{\pi}\mu [\si{\watt/\meter^{2}}]
                \end{eqnarray*}

              \item $U(\theta)$ $=$ $B_{0}\cos^{3}{\theta}$. O cálculo
                é análogo ao item anterior.

                \begin{eqnarray*}
                  14 & = &\int^{2\pi}_{0}\int^{\frac{\pi}{2}}_{0}
                           B_{0}\cos^{3}{\theta}\sin{\theta} d\theta d\phi \\
                  14 & = & 2B_{0}\pi \int^{\frac{\pi}{2}}_{0} \cos^{3}{\theta}\sin{\theta} d\theta
                \end{eqnarray*}
                
                Fazendo a seguinte substituição de variável

                $ \left\{\begin{array}{l}
                           u = \cos{\theta} \\
                           -du = \sin{\theta} d\theta
                         \end{array}\right.
                       $

                       e dos limites, temos

                       \begin{eqnarray*}
                         14 & = & (-1)2B_{0}\pi \int^{0}_{1}u^{3} du \\
                            & = & 2B_{0}\pi \int^{1}_{0}u^{3} du \\                  
                            & = & 2B_{0}\pi \left(\dfrac{u^{4}}{4}\right)^{1}_{0} \\
                         14 & = & \dfrac{B_{0}\pi}{2} \\
                         B_{0} & = & \dfrac{28}{\pi}
                       \end{eqnarray*}

                       A densidade de potência é dado por
                       $W_{rad} = \dfrac{U(\theta,\phi)}{r^{2}}$, com
                       $U(\theta,\phi) =
                       \dfrac{28}{\pi}\cos^{3}{\theta}$, para seu
                       valor máximo temos $\theta = k\pi$, com
                       $k = 2n$ e $n \in \mathbb{Z}$, assim
                       $\cos^{3}{\theta}$ $=$ $1$.

                       \begin{eqnarray*}
                         W_{M} & = & \dfrac{28}{\pi}\left(
                                     \dfrac{1}{10^{3}}\right)^{2} \\
                         W_{M} & = & \dfrac{28}{\pi}\mu [\si{\watt/\meter^{2}}]
                       \end{eqnarray*}

                     \end{enumerate}

                   \item O ângulo sólido de feixe $\Omega_{A}$ exato e
                     aproximado.

                     \begin{enumerate}[i.]
                     \item Seja $\Omega_{A}$ $=$
                       $\displaystyle \int^{2\pi}_{0}\int^{\pi}_{0}$
                       $U_{n} \, d\Omega_{A}$, com
                       $U_{n} =
                       \dfrac{U(\theta,\phi)}{\left|U(\theta,\phi)_{M}\right|}$
                       $\Rightarrow$
                       $U_{n} = \dfrac{\frac{21}{\pi}}{\frac{21}{\pi}}
                       \cos^{2}{\theta}$ e
                       $d\Omega_{A} = \sin{\theta} \, d\theta \,
                       d\phi$, assim temos:

                       \begin{eqnarray*}
                         \Omega_{A} & = &
                                          \int^{2\pi}_{0}\int^{\pi}_{0}
                                          \cos^{2}{\theta}\sin{\theta}
                                          \, d\theta \, d\phi \\
                                    & = & 2\pi \int^{\pi}_{0}
                                          \cos^{2}{\theta}\sin{\theta}
                                          \, d\theta
                       \end{eqnarray*}

                       Fazendo a seguinte substituição de variável


                       $ \left\{\begin{array}{l}
                                  u = \cos{\theta} \\
                                  -du = \sin{\theta} d\theta
                                \end{array}\right.
                              $

                              e dos limites, temos

                      \begin{eqnarray*}
                        \Omega_{A} & = & (-2\pi) \int^{-1}_{1}u^{2} \, du \\
                                   & = & 2\pi \, \left(\dfrac{u^{3}}{3}\right)^{1}_{-1} \\
                        \Omega_{A} & = & \dfrac{4\pi}{3} \si{\steradian}
                      \end{eqnarray*}

                    \item Com o cálculo análogo ao item anterior temos
                      $U_{n} = \dfrac{\frac{28}{\pi}}{\frac{28}{\pi}}
                      \cos^{3}{\theta}$ e
                      $d\Omega_{A} = \sin{\theta} \, d\theta \,
                      d\phi$, assim temos:

                       \begin{eqnarray*}
                         \Omega_{A} & = &
                                          \int^{2\pi}_{0}\int^{\pi}_{0}
                                          \cos^{3}{\theta}\sin{\theta}
                                          \, d\theta \, d\phi \\
                                    & = & 2\pi \int^{\pi}_{0}
                                          \cos^{3}{\theta}\sin{\theta}
                                          \, d\theta
                       \end{eqnarray*}

                       Fazendo a seguinte substituição de variável


                       $ \left\{\begin{array}{l}
                                  u = \cos{\theta} \\
                                  -du = \sin{\theta} d\theta
                                \end{array}\right.
                              $

                              e dos limites, temos

                      \begin{eqnarray*}
                        \Omega_{A} & = & (-2\pi) \int^{-1}_{1}u^{3} \, du \\
                                   & = & 2\pi \, \left(\dfrac{u^{4}}{4}\right)^{1}_{-1} \\
                        \Omega_{A} & = & 0 \si{\steradian}
                      \end{eqnarray*}

                    \item A área aproximada é dado em termos dos
                      ângulos entre os pontos de LFMP do lóbulo
                      principal nos dois principais planos
                      $(\theta,\phi)$. Assim temos:
                      
                      \begin{enumerate}[1.]
                      \item
                        $U(\theta) = \dfrac{21}{\pi}\cos^{2}{\theta}$
                        $\Rightarrow$
                        $\dfrac{21}{\pi}\cos^{2}{\theta_{h}}$ $=$
                        $\dfrac{1}{2}$ $\Rightarrow$
                        $\cos{\theta_{h}}$ $=$
                        $\sqrt{\dfrac{\pi}{42}}$ $\Rightarrow$
                        $\theta_{h} =
                        \cos^{-1}{\sqrt{\dfrac{\pi}{42}}}$
                        $\Rightarrow$
                        $\theta_{h} \approx 1,293 \si{\radian}$
                        . Assim a LFMP na direção de $\theta$ é dado
                        por $2\theta_{h} \approx 2,587 \si{\radian}$.
                        Por simetria, já que o padrão de feixe
                        independe de $\phi$,
                        $\Theta_{1r} = \Theta_{2r}$. Assim podemos
                        calcular
                        $\Omega_{A} \approx \Theta_{1} \Theta_{2}$
                        $\approx$ $2,587^{2}$ $\approx$
                        $6,695 \si{\steradian}$
                      \item
                        $U(\theta) = \dfrac{28}{\pi}\cos^{3}{\theta}$
                        $\Rightarrow$
                        $\dfrac{28}{\pi}\cos^{3}{\theta_{h}}$ $=$
                        $\dfrac{1}{2}$ $\Rightarrow$
                        $\cos{\theta_{h}}$ $=$
                        $\sqrt[3]{\dfrac{\pi}{56}}$ $\Rightarrow$
                        $\theta_{h} =
                        \cos^{-1}{\sqrt[3]{\dfrac{\pi}{56}}}$
                        $\Rightarrow$
                        $\theta_{h} \approx 1,177 \si{\radian}$.
                        Assim a LFMP na direção de $\theta$ é dado por
                        $2\theta_{h} \approx 2,355 \si{\radian}$. Assim podemos
                        calcular
                        $\Omega_{A} \approx \Theta_{1} \Theta_{2}$
                        $\approx$ $2,355^{2}$ $\approx$
                        $5,55 \si{\steradian}$
                      \end{enumerate}

                    \end{enumerate}
                     
                  \end{enumerate}
